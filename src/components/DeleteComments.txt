In order to delete a comment direct form the CLI:
db.docs.updateOne(
  { email: "mohamedbnshi@gmail.com", "docs.title": "Mo testar ghghh ds" },
  { $pull: { "docs.$.comments": { _id: ObjectId("68f829cef43476492bc21e6e") } } }
);

In order to delete all comments at once:
db.docs.updateOne(
  { email: "mohamedbnshi@gmail.com", "docs.title": "Mo testar ghghh ds" },
  { $pull: { "docs.$.comments": {} } }   // {} matches all -> removes all comments
);
